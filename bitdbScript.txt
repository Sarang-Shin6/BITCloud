drop database bitdb;
create database bitdb;
use bitdb;



CREATE TABLE JobCategory (
                JobCategoryID INT AUTO_INCREMENT NOT NULL,
                Description VARCHAR(50) NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (JobCategoryID)
);


CREATE TABLE Status (
                StatusID INT AUTO_INCREMENT NOT NULL,
                Description VARCHAR(20) NOT NULL,
                PRIMARY KEY (StatusID)
);


CREATE TABLE Client (
                ClientID INT AUTO_INCREMENT NOT NULL,
                FirstName VARCHAR(15) NOT NULL,
                LastName VARCHAR(15) NOT NULL,
                Phone VARCHAR(10) NOT NULL,
                Email VARCHAR(30) NOT NULL,
                Gender CHAR(1) NOT NULL,
                Password VARCHAR(40) NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (ClientID)
);


CREATE TABLE Client_log (
                ClientLogID INT AUTO_INCREMENT NOT NULL,
                ClientID INT NOT NULL,
                FirstName VARCHAR(15) NOT NULL,
                LastName VARCHAR(15) NOT NULL,
                Phone VARCHAR(10) NOT NULL,
                Email VARCHAR(30) NOT NULL,
                Password VARCHAR(40) NOT NULL,
                Gender CHAR(1) NOT NULL,
                Action CHAR(3) NOT NULL,
                ChangeDate DATE NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (ClientLogID)
);


CREATE TABLE Location (
                LocationID INT AUTO_INCREMENT NOT NULL,
                City VARCHAR(10) NOT NULL,
                Description VARCHAR(30) NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (LocationID)
);


CREATE TABLE Company (
                CompanyID INT AUTO_INCREMENT NOT NULL,
                Name VARCHAR(30) NOT NULL,
                Street VARCHAR(20) NOT NULL,
                Suburb VARCHAR(12) NOT NULL,
                State VARCHAR(4) NOT NULL,
                Postcode VARCHAR(4) NOT NULL,
                Branch VARCHAR(20) NOT NULL,
                Description VARCHAR(50) NOT NULL,
                ClientID INT NOT NULL,
                LocationID INT NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (CompanyID)
);


CREATE TABLE Company_log (
                CompanyLogID INT AUTO_INCREMENT NOT NULL,
                CompanyID INT NOT NULL,
                Name VARCHAR(30) NOT NULL,
                Street VARCHAR(20) NOT NULL,
                Suburb VARCHAR(12) NOT NULL,
                State VARCHAR(4) NOT NULL,
                Postcode VARCHAR(4) NOT NULL,
                Branch VARCHAR(20) NOT NULL,
                Description VARCHAR(50) NOT NULL,
                ClientID INT NOT NULL,
                LocationID INT NOT NULL,
                Action CHAR(3) NOT NULL,
                ChangeDate DATE NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (CompanyLogID)
);


CREATE TABLE Staff (
                StaffID INT AUTO_INCREMENT NOT NULL,
                FirstName VARCHAR(15) NOT NULL,
                LastName VARCHAR(15) NOT NULL,
                Street VARCHAR(20) NOT NULL,
                Suburb VARCHAR(12) NOT NULL,
                State VARCHAR(4) NOT NULL,
                Postcode VARCHAR(4) NOT NULL,
                Email VARCHAR(30) NOT NULL,
                Phone VARCHAR(10) NOT NULL,
                AccName VARCHAR(30) NOT NULL,
                AccBSB VARCHAR(6) NOT NULL,
                AccNumber VARCHAR(9) NOT NULL,
                Gender CHAR(1) NOT NULL,
                Password VARCHAR(40) NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (StaffID)
);


CREATE TABLE Staff_log (
                StaffLogID INT AUTO_INCREMENT NOT NULL,
                StaffID INT NOT NULL,
                FirstName VARCHAR(15) NOT NULL,
                LastName VARCHAR(15) NOT NULL,
                Street VARCHAR(20) NOT NULL,
                Suburb VARCHAR(12) NOT NULL,
                State VARCHAR(4) NOT NULL,
                Postcode VARCHAR(4) NOT NULL,
                Email VARCHAR(30) NOT NULL,
                Phone VARCHAR(10) NOT NULL,
                AccName VARCHAR(30) NOT NULL,
                AccBSB VARCHAR(6) NOT NULL,
                AccNumber VARCHAR(9) NOT NULL,
                Gender CHAR(1) NOT NULL,
                Password VARCHAR(40) NOT NULL,
                Action CHAR(3) NOT NULL,
                ChangeDate DATE NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (StaffLogID)
);


CREATE TABLE Coordinator (
                CoordinatorID INT NOT NULL,
                Admin TINYINT NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (CoordinatorID)
);


CREATE TABLE Contractor (
                ContractorID INT NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (ContractorID)
);


CREATE TABLE ContractorAvailability (
                ContractorAvailabilityID INT AUTO_INCREMENT NOT NULL,
                Date DATE NOT NULL,
                StartTime TIME NOT NULL,
                Hours TINYINT NOT NULL,
                ContractorID INT NOT NULL,
		Active BIT NOT NULL,
                PRIMARY KEY (ContractorAvailabilityID)
);


CREATE TABLE ContractorLocation (
                ContractorLocationID INT AUTO_INCREMENT NOT NULL,
                LocationID INT NOT NULL,
                ContractorID INT NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (ContractorLocationID)
);


CREATE TABLE ContractorSkill (
                ContractorSkillID INT AUTO_INCREMENT NOT NULL,
                JobCategoryID INT NOT NULL,
                ContractorID INT NOT NULL,
                Active TINYINT NOT NULL,
                PRIMARY KEY (ContractorSkillID)
);


CREATE TABLE Job (
                JobID INT AUTO_INCREMENT NOT NULL,
                CompanyID INT NOT NULL,
                ClientID INT NOT NULL,
                StatusID INT NOT NULL,
                JobCategoryID INT NOT NULL,
                StartDate DATE,
                Duration INT,
                PreferredEndDate DATE NOT NULL,
                EstimatedDuration INT,
                Description VARCHAR(100) NOT NULL,
                Feedback VARCHAR(100),
                LoggedKm INT,
                JobRating TINYINT,
                ContractorID INT,
                CoordinatorID INT,
                Active TINYINT NOT NULL,
                PRIMARY KEY (JobID)
);


CREATE TABLE Job_log (
                JobLogID INT AUTO_INCREMENT NOT NULL,
                JobID INT NOT NULL,
                CompanyID INT NOT NULL,
                ClientID INT NOT NULL,
                StatusID INT NOT NULL,
                JobCategoryID INT NOT NULL,
                StartDate DATE,
                Duration INT,
                PreferredEndDate DATE NOT NULL,
                EstimatedDuration INT,
                Description VARCHAR(100) NOT NULL,
                Feedback VARCHAR(100),
                LoggedKm INT,
                JobRating TINYINT,
                ContractorID INT,
                CoordinatorID INT,
		Active TINYINT NOT NULL,
                Action CHAR(3) NOT NULL,
                ChangeDate DATE NOT NULL,
                PRIMARY KEY (JobLogID)
);

CREATE TABLE JobRejection (
                JobRejectionID INT AUTO_INCREMENT NOT NULL,
                ContractorID INT NOT NULL,
                JobID INT NOT NULL,
                PRIMARY KEY (JobRejectionID)
);

CREATE PROCEDURE usp_Cloud_ClientGetJobs(in clientID INT)
BEGIN
    select job.StartDate, job.JobID, staff.FirstName, staff.LastName, job.description, jobcategory.Description, status.Description As Status
	from job
    left join staff on job.ContractorID = staff.StaffID
    inner join jobcategory on job.JobCategoryID = jobcategory.JobCategoryID
    inner join status on job.StatusID = status.StatusID
    where job.ClientID = clientID
    and job.Active = 1
    order by job.StartDate;
END;;

CREATE PROCEDURE usp_Cloud_ContractorAssignRequest(in conID INT)
BEGIN
    select job.JobID, job.PreferredEndDate, client.FirstName, client.LastName, client.Phone, company.Name, company.Branch, company.Street, company.Suburb, company.State, company.Postcode, job.description, jobcategory.Description
    from job, company, jobcategory, client
    where job.CompanyID = company.CompanyID
    and job.JobCategoryID = jobcategory.JobCategoryID
    and job.ContractorID = conID
    and job.StatusID = 3
    and job.ClientID = client.ClientID;
END;;

    and job.StatusID in (4, 5, 6, 7, 8)
    and status.StatusID = job.StatusID
    and job.Active = 1;
END;;

-- get contractor based on location and skill --

drop PROCEDURE if exists usp_GetContractors;;
create procedure usp_GetContractors(in jobcatid int, in locid int)
begin
select 			c.contractorid,
				s.firstname,
                s.lastname,
				cs.jobcategoryid, 
                jc.description as jobcat,
            	cl.locationid, 
                l.description as loc,
            	AVG(j.jobrating) as AverageRating
			
                
from 			contractor as c
				inner join contractorskill as cs on c.ContractorID = cs.ContractorID
				inner join contractorlocation as cl on c.ContractorID = cl.ContractorID
				inner join job as j on c.ContractorID = j.ContractorID
                inner join staff as s on s.StaffID = c.ContractorID
                inner join location as l on l.LocationID = cl.LocationID
                inner join jobcategory as jc on jc.JobCategoryID = cs.JobCategoryID
where 			c.active = 1
and 			cs.jobcategoryid = jobcatid
and 			cl.locationid = locid
Group by 		c.ContractorID
order by 		AverageRating DESC;

end;;

-- get totalhours --

drop PROCEDURE if exists usp_GetContractorsHours;;
create procedure usp_GetContractorsHours(in estimatedduration int, in duedate date)
begin
select c.contractorid, sum(ca.Hours) as TotalHours
from contractor as c
inner join contractoravailability as ca on c.ContractorID = ca.ContractorID
where ca.Date BETWEEN now() and duedate
group by ContractorID
HAVING TotalHours >= estimatedduration;

end;;


-- get totaljobs

drop PROCEDURE if exists usp_GetContractorsTotalJobs;;
create procedure usp_GetContractorsTotalJobs()
begin

select distinct j.contractorid, count(*) as totalJobs
from job as j
where j.active = 1
and j.statusid between 3 and 5
and j.contractorid is not null
group by contractorid
having totalJobs < 2;



end;;

-- get jobrejection

drop PROCEDURE if exists usp_GetRejections;;
create procedure usp_GetRejections(in conid int, in jobid int)
begin

select j.contractorid, j.jobid 
from jobrejection as j
where		j.contractorid = conid
and		j.jobid = jobid;


end;;
